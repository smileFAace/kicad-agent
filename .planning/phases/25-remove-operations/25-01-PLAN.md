```python
def get_wire_by_uuid(self, uuid: str) -> Optional[Any]:
    """Find a wire (Connection with type='wire') by its UUID."""
    from kiutils.items.schitems import Connection
    for item in self._parse_result.kiutils_obj.graphicalItems:
        if isinstance(item, Connection) and item.type == "wire":
            if item.uuid == uuid:
                return item
    return None

def get_label_by_uuid(self, uuid: str, label_type: str = "local") -> Optional[Any]:
    """Find a label by UUID and type (local/global/hierarchical)."""
    sch = self._parse_result.kiutils_obj
    if label_type == "global":
        for label in sch.globalLabels:
            if label.uuid == uuid:
                return label
    elif label_type == "hierarchical":
        for label in sch.hierarchicalLabels:
            if label.uuid == uuid:
                return label
    else:
        for label in sch.labels:
            if label.uuid == uuid:
                return label
    return None

def get_junction_by_uuid(self, uuid: str) -> Optional[Any]:
    """Find a junction by UUID."""
    for jct in self._parse_result.kiutils_obj.junctions:
        if jct.uuid == uuid:
            return jct
    return None

def get_no_connect_by_uuid(self, uuid: str) -> Optional[Any]:
    """Find a no-connect marker by UUID."""
    for nc in self._parse_result.kiutils_obj.noConnects:
        if nc.uuid == uuid:
            return nc
    return None
```

```python
def get_adjacent_wires(self, wire: Any, tolerance: float = 0.0001) -> list:
    """Find all other wires sharing endpoints with the given wire.

    Two wires are adjacent if one's start or end point matches the other's
    start or end point within the specified tolerance.

    Args:
        wire: A kiutils Connection object with type='wire'.
        tolerance: Maximum coordinate distance to consider endpoints matching (default 0.0001).

    Returns:
        List of adjacent Connection objects (excluding the wire itself).
    """
    from kiutils.items.schitems import Connection
    if not hasattr(wire, 'points') or len(wire.points) < 2:
        return []

    sx, sy = wire.points[0].X, wire.points[0].Y
    ex, ey = wire.points[1].X, wire.points[1].Y

    def _coords_match(ax, ay, bx, by) -> bool:
        return abs(ax - bx) <= tolerance and abs(ay - by) <= tolerance

    adjacent = []
    for item in self._parse_result.kiutils_obj.graphicalItems:
        if item is wire:
            continue
        if isinstance(item, Connection) and item.type == "wire":
            if len(item.points) >= 2:
                isx, isy = item.points[0].X, item.points[0].Y
                iex, iey = item.points[1].X, item.points[1].Y
                if (
                    _coords_match(sx, sy, isx, isy) or _coords_match(sx, sy, iex, iey)
                    or _coords_match(ex, ey, isx, isy) or _coords_match(ex, ey, iex, iey)
                ):
                    adjacent.append(item)
    return adjacent
```
