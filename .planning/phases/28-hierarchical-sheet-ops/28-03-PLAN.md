**add_sheet_pin(op, ir, file_path) implementation:**

```python
def add_sheet_pin(op: Any, ir: SchematicIR, file_path: Path) -> dict[str, Any]:
    """Add a hierarchical pin to an existing sheet instance.

    Validates pin_name matches a hierarchical label in the child sheet
    using EXACT == comparison (PITFALL 1: case-sensitive, no fuzzy matching).
    Validates pin position is on the sheet boundary (PITFALL 11).
    """
    import uuid
    from kiutils.schematic import HierarchicalPin
    from kiutils.items.common import Position, Effects, Font

    sch = ir.schematic

    # 1. Find the target sheet by UUID
    target_sheet = None
    for sheet in sch.sheets:
        if sheet.uuid == op.sheet_uuid:
            target_sheet = sheet
            break

    if target_sheet is None:
        raise ValueError(f"Sheet with UUID '{op.sheet_uuid}' not found in schematic")

    # 2. Validate pin position is on sheet boundary (PITFALL 11)
    sheet_x = target_sheet.position.X
    sheet_y = target_sheet.position.Y
    sheet_w = target_sheet.width
    sheet_h = target_sheet.height
    tolerance = 0.01  # mm

    pin_x = op.position.x
    pin_y = op.position.y

    # Check if position is on any boundary edge
    on_left = abs(pin_x - sheet_x) <= tolerance
    on_right = abs(pin_x - (sheet_x + sheet_w)) <= tolerance
    on_top = abs(pin_y - sheet_y) <= tolerance
    on_bottom = abs(pin_y - (sheet_y + sheet_h)) <= tolerance
    in_y_range = sheet_y - tolerance <= pin_y <= sheet_y + sheet_h + tolerance
    in_x_range = sheet_x - tolerance <= pin_x <= sheet_x + sheet_w + tolerance

    is_on_boundary = (
        (on_left and in_y_range) or
        (on_right and in_y_range) or
        (on_top and in_x_range) or
        (on_bottom and in_x_range)
    )

    if not is_on_boundary:
        # Snap to nearest boundary edge
        edges = [
            (abs(pin_x - sheet_x), sheet_x, pin_y, "left"),         # left edge
            (abs(pin_x - (sheet_x + sheet_w)), sheet_x + sheet_w, pin_y, "right"),  # right edge
            (abs(pin_y - sheet_y), pin_x, sheet_y, "top"),          # top edge
            (abs(pin_y - (sheet_y + sheet_h)), pin_x, sheet_y + sheet_h, "bottom"),  # bottom edge
        ]
        # Find closest edge
        edges.sort(key=lambda e: e[0])
        _, snapped_x, snapped_y, edge_name = edges[0]
        # Clamp to edge
        if edge_name in ("left", "right"):
            snapped_y = max(sheet_y, min(sheet_y + sheet_h, pin_y))
        else:
            snapped_x = max(sheet_x, min(sheet_x + sheet_w, pin_x))
        pin_x = snapped_x
        pin_y = snapped_y

    # 3. Parse child sheet and validate pin_name matches a hierarchical label (PITFALL 1)
    child_file_name = target_sheet.fileName.value if target_sheet.fileName else ""
    if not child_file_name:
        raise ValueError("Target sheet has no fileName property")

    child_path = file_path.resolve().parent / child_file_name
    if not child_path.exists():
        raise ValueError(f"Child sheet file not found: {child_path}")

    child_result = parse_schematic(child_path)
    child_ir = SchematicIR(_parse_result=child_result)

    # EXACT == comparison (PITFALL 1: case-sensitive, no .lower(), no .strip())
    child_label_names = [lbl.text for lbl in child_ir.schematic.hierarchicalLabels if lbl.text]
    if op.pin_name not in child_label_names:
        raise ValueError(
            f"Pin name '{op.pin_name}' does not match any hierarchical label in "
            f"child sheet '{child_file_name}'. Available labels: {child_label_names}. "
            f"Matching is case-sensitive and exact (PITFALL 1)."
        )

    # 4. Create the HierarchicalPin
    pin_uuid = str(uuid.uuid4())
    pin = HierarchicalPin(
        name=op.pin_name,
        connectionType=op.connection_type,
        position=Position(X=pin_x, Y=pin_y),
        effects=Effects(font=Font()),
        uuid=pin_uuid,
    )

    # 5. Append to sheet's pin list
    target_sheet.pins.append(pin)

    # 6. Record mutation
    ir._record_mutation("add_sheet_pin", {
        "sheet_uuid": op.sheet_uuid,
        "pin_name": op.pin_name,
        "pin_uuid": pin_uuid,
        "connection_type": op.connection_type,
    })

    return {
        "pin_name": op.pin_name,
        "pin_uuid": pin_uuid,
        "connection_type": op.connection_type,
        "position": [pin_x, pin_y],
        "sheet_uuid": op.sheet_uuid,
    }
```

**Key safety constraints enforced:**
- PITFALL 1: Exact `==` comparison for pin_name vs child label names. No `.lower()`, no `.strip()`, no fuzzy matching.
- PITFALL 11: Pin position validated on sheet boundary. Snaps to nearest edge if off-boundary.
- PITFALL 2: Child file path resolved relative to parent file directory (`file_path.resolve().parent / child_file_name`).
