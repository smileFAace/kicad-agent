**navigate_hierarchy(op, ir, file_path) implementation:**

1. Build a tree by recursively walking `sch.sheets`:

```python
def navigate_hierarchy(op: Any, ir: SchematicIR, file_path: Path) -> dict[str, Any]:
    """Return the hierarchical sheet tree starting from the target file."""
    root_info = _build_sheet_node(ir, file_path, depth=0, max_depth=op.max_depth, visited=set())
    return {"hierarchy": root_info}
```

2. Internal recursive function:

```python
def _build_sheet_node(ir: SchematicIR, file_path: Path, depth: int, max_depth: int, visited: set[str]) -> dict[str, Any]:
    sch = ir.schematic
    node = {
        "file_name": str(file_path.name),
        "file_path": str(file_path),
        "uuid": sch.uuid,
        "components": len(sch.schematicSymbols),
        "hierarchical_labels": [lbl.text for lbl in sch.hierarchicalLabels if lbl.text],
        "sheets": [],
    }

    if max_depth >= 0 and depth >= max_depth:
        return node

    for sheet in sch.sheets:
        sheet_file = sheet.fileName.value if sheet.fileName else ""
        if not sheet_file:
            continue

        sub_path = file_path.resolve().parent / sheet_file

        # Cycle detection (SHEET-06)
        sub_path_str = str(sub_path.resolve())
        if sub_path_str in visited:
            node["sheets"].append({
                "file_name": sheet_file,
                "uuid": sheet.uuid,
                "error": "circular_reference_detected",
            })
            continue

        if not sub_path.exists():
            node["sheets"].append({
                "file_name": sheet_file,
                "uuid": sheet.uuid,
                "error": "file_not_found",
                "pin_count": len(sheet.pins),
            })
            continue

        # Parse the sub-sheet and recurse
        try:
            sub_result = parse_schematic(sub_path)
            sub_ir = SchematicIR(_parse_result=sub_result)
            child_node = _build_sheet_node(
                sub_ir, sub_path,
                depth=depth + 1,
                max_depth=max_depth,
                visited=visited | {sub_path_str},
            )
            child_node["uuid"] = sheet.uuid
            child_node["pin_count"] = len(sheet.pins)
            child_node["pin_names"] = [p.name for p in sheet.pins if p.name]
            node["sheets"].append(child_node)
        except Exception as exc:
            node["sheets"].append({
                "file_name": sheet_file,
                "uuid": sheet.uuid,
                "error": f"parse_error: {exc}",
            })

    return node
```
